// pca_model.h

#ifndef PCA_MODEL_H

#define PCA_MODEL_H

 

#define PCA_N_FEATURES 5

#define PCA_N_COMPONENTS 2

 

extern const float pca_components[PCA_N_COMPONENTS][PCA_N_FEATURES];

extern const float pca_mean[PCA_N_FEATURES];

 

static inline float pca_anomaly_score(const float *x) {

float centered[PCA_N_FEATURES];

for (int i = 0; i < PCA_N_FEATURES; i++)

centered[i] = x[i] - pca_mean[i];

 

float recon[PCA_N_FEATURES] = {0};

for (int i = 0; i < PCA_N_COMPONENTS; i++) {

float proj = 0;

for (int j = 0; j < PCA_N_FEATURES; j++)

proj += centered[j] * pca_components[i][j];

for (int j = 0; j < PCA_N_FEATURES; j++)

recon[j] += proj * pca_components[i][j];

}

 

float err = 0;

for (int i = 0; i < PCA_N_FEATURES; i++) {

float diff = centered[i] - recon[i];

err += diff * diff;

}

return err;

}

 

#endif

📎 Appendix A: LLM Prompt Template (for rank_features.py)

 

You are an expert in industrial anomaly detection. Given feature statistics comparing nominal vs powered-off operation:

 

FEATURES:

- accZ__fft_coefficient__coeff_3__attr_abs: μ₀=4.3, σ₀=0.5; μ₁=0.2, σ₁=0.05

- temp__mean_change: μ₀=0.05, σ₀=0.03; μ₁=-0.01, σ₁=0.02

...

 

Rank the top 5 most important features for detecting *future mechanical anomalies* (e.g., imbalance, bearing wear). Prioritize:

1. Large separation (|μ₀−μ₁|),

2. Low intra-class variance (small σ),

3. Physical interpretability.

 

Output ONLY a JSON list:

[{"name":"...", "axis":int, "type":"...", "bin"?:int, "window"?:int, "score":float, "reason":"..."}]
